"use client";

import { useEffect } from "react";
import { useRouter } from "next/navigation";
import {BadgeCheck} from 'lucide-react' 
import { useUser } from "@/context/UserContext";

export default function SuccessPage() {
  const router = useRouter();
  const { user } = useUser();

  useEffect(() => {
    const timer = setTimeout(() => {
      router.push(`/${user.slug}/dashboard/reservas`); // Redirección automática
    }, 2000);

    return () => clearTimeout(timer);
  }, [router]);

  return (
    <div className="min-h-screen bg-brand-black flex flex-col items-center justify-center p-4 rounded-2xl">
      <div className="bg-brand-white/5 p-8 rounded-2xl border border-brand-gold/20 text-center max-w-md w-full animate-in fade-in zoom-in duration-500">
        <div className="flex justify-center mb-6">
          <BadgeCheck  className="w-20 h-20 text-brand-gold animate-bounce" />
        </div>
        
        <h1 className="text-3xl font-bold text-brand-white mb-2 italic uppercase">
          ¡Pago Exitoso!
        </h1>
        <p className="text-brand-gray/80 mb-8">
          Tu reserva en Arena Prometeo ha sido confirmada.
        </p>

        <div className="space-y-4">
           <div className="h-1 w-full bg-brand-black rounded-full overflow-hidden">
            <div className="h-full bg-brand-gold animate-progress-bar"></div>
          </div> 

          <p className="text-xs text-brand-gold uppercase tracking-widest font-semibold">
            Redirigiendo a tus reservas...
          </p>
            <div className="mt-3 w-40 h-[2px] bg-brand-white/10 overflow-hidden rounded-full mx-auto">
                    <div className="h-full bg-brand-gold w-1/4 animate-loading-slide"></div>
                </div>
        </div>
      </div>
    </div>
  );
}
